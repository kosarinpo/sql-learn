For the database with money transactions being recorded
not more than once a day, calculate the total cash balance of
all buy-back centers.

SELECT
    SUM(Data.inc)
FROM
    (
        SELECT
            Income_o.point
          , Income_o.inc
        FROM
            Income_o
        UNION ALL
        SELECT
            Outcome_o.point
          , (-1) * Outcome_o.out AS inc
        FROM
            Outcome_o
    ) AS Data
